@startuml

skinparam classAttributeIconSize 0

class User {
    -isAuthenticated : bool
    -username : string
    -password : string
    -hasTwoFactorAuthenticaton : bool
    +isAuthenticated() : bool
    -roles : List<Role>
    +hasPermission(Permission) : bool
    -contactInformation : ContactInformation
    -birthday : Date
    -doNotVisit : bool
    -personalCalendar : PersonalCalendar
    -ward : Ward
}

class ContactInformation {
  -name : string
  -phoneNumbers : List<PhoneNumber>
  -addresses : List<Address>
  -emailAddress : string
}

class PhoneNumber {
  -number : string
  -type : PhoneType
}

enum PhoneType {
    Home
    Cell
    Fax
    Work
}

class Address {
  -addressLine1 : string
  -addressLine2 : string
  -addressLine3 : string
  -addressLine4 : string
  -addressLine5 : string
  -city : string
  -stateOrProvince : string
  -postalCode : string
  -county : string
  -country : string
  -addressType : AddressType
}

enum AddressType {
    Home
    Work
}

class Role {
    -name : string
    -permissions : Set<Permission>
}

enum Permission {
    ViewMemberContactInformation
    ResetPassword
    CreateAssignment
    UpdateAssignment
    DeleteAssignment
    CreateCompanionship
    UpdateCompanionship
    DeleteCompanionship
    ...TBD
}

class Stake {
    -wards : List<Ward>
    -name : string
}

class Ward {
    -members : List<Users>
    -stake : Stake
    -name : string
}

Ward o-right- User : > has 0..*
Stake o-up- Ward : > has 1..*
User o-- Role : > has 1..*
Role *-- Permission : > has 1..*
User *-- ContactInformation : > has 1..*
ContactInformation *-- PhoneNumber : > has 1..*
PhoneNumber *-- PhoneType : > has 1
ContactInformation *-- Address : > has 1..*
Address *-- AddressType  : > has 1

@enduml
